\documentclass[border=8pt]{standalone}
\usepackage[UTF8]{ctex}
\usepackage{tikz}
\usepackage{amsmath,amssymb}
\usetikzlibrary{arrows.meta,calc}
\definecolor{cBlue}{RGB}{37,99,235}
\definecolor{cOrange}{RGB}{234,88,12}
\definecolor{cGray}{RGB}{100,116,139}
\definecolor{cGreen}{RGB}{22,163,74}
\definecolor{cRed}{RGB}{220,38,38}
\begin{document}
\begin{tikzpicture}[scale=1.9,>=Stealth,line width=0.9pt]
  \coordinate (O) at (0,0);
  % rotation axis n
  \draw[cRed,line width=1.3pt,->] (O)--(0,1.75) node[above] {$\hat n$ （转轴）};
  % the circle that v sweeps (horizontal, lifted)
  \draw[cGray,dashed] (0,0.95) ellipse (1.15 and 0.40);
  % v (start) and v' (rotated)
  \draw[cBlue,line width=1.3pt,->] (O)--(1.15,0.95);
  \node[cBlue,right] at (1.15,0.92) {$v$};
  \draw[cGreen,line width=1.3pt,->] (O)--(-0.66,1.30);
  \node[cGreen,left] at (-0.66,1.33) {$v'$};
  % the swept angle phi on the ellipse
  \draw[cOrange,line width=1.0pt,->] (0.92,1.10) arc (20:163:0.92 and 0.32);
  \node[cOrange] at (0.15,1.55) {$\varphi$};
  \fill[cGray] (O) circle (0.03);
  \node[align=center] at (0,-0.65)
    {$v'=q\,v\,q^{-1},\qquad q=\cos\tfrac{\varphi}{2}+\sin\tfrac{\varphi}{2}\,\hat n$};
  \node[cGray,align=center] at (0,-1.05)
    {\footnotesize 旋转 $\varphi$，四元数里却是\textbf{半角} $\varphi/2$——转 $360^\circ$ 时 $q$ 变成 $-q$。};
\end{tikzpicture}
